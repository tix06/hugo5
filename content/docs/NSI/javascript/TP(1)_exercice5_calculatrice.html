On vous donne l'interface d'une calculatrice déjà mise en page (boutons 0–9, +, −, ×, ÷, =, C) et un écran d'affichage une ligne (style CASIO).
Votre mission : compléter le script pour la faire fonctionner avec des réels.

0
💡 Pistes :
  • Créez deux variables : let operande = ""; pour construire le nombre en cours, et let operateur = ""; pour mémoriser l'opération choisie (+, −, ×, ÷).
  • Créez aussi let premierNombre = 0; pour stocker la valeur avant l'opérateur.
  • Quand on clique sur un chiffre, on ajoute ce chiffre à operande et on met à jour l'écran : ecran.textContent = operande;
  • Quand on clique sur un opérateur, on mémorise premierNombre = parseFloat(operande);, on note l'opérateur, et on remet operande = ""; à vide pour saisir le second nombre.
  • Quand on clique sur =, on calcule le résultat selon l'opérateur mémorisé (avec if / else if) et on l'affiche.
  • C remet tout à zéro.
  • parseFloat("3.14") convertit une chaîne en nombre réel.

// ========================================
// CORRECTION COMPLÈTE

let operande      = "";
let premierNombre = null;
let operateur     = "";

function clicChiffre(chiffre) {
    // On évite deux points décimaux
    if (chiffre === '.' && operande.includes('.')) return;
    operande += chiffre;
    ecran.textContent = operande;
}

function clicOperateur(op) {
    if (operande === "") return; // rien à faire si pas de nombre saisi
    premierNombre = parseFloat(operande);
    operateur     = op;
    operande      = "";
    ecran.textContent = "0";
}

function clicEgal() {
    if (premierNombre === null || operande === "") return;
    const deuxiemeNombre = parseFloat(operande);
    let resultat;

    if      (operateur === '+') resultat = premierNombre + deuxiemeNombre;
    else if (operateur === '-') resultat = premierNombre - deuxiemeNombre;
    else if (operateur === '*') resultat = premierNombre * deuxiemeNombre;
    else if (operateur === '/') {
        if (deuxiemeNombre === 0) {
            ecran.textContent = "Erreur";
            return;
        }
        resultat = premierNombre / deuxiemeNombre;
    }

    ecran.textContent = resultat;
    // Prépare un éventuel enchaînement de calculs
    operande      = String(resultat);
    premierNombre = null;
    operateur     = "";
}

function clicClear() {
    operande      = "";
    premierNombre = null;
    operateur     = "";
    ecran.textContent = "0";
}
